useSwitch/index.js

/** 
 * useSwitch
 * @module useSwitch
*/
import { useState, useMemo, useRef, useEffect } from 'react';



/**
 * 控制开关
 * 
 * [Demo-CodeSandbox]{@link https://codesandbox.io/s/useswitch-qd7dr?file=/index.js}
 * 
 * @param  {boolean} [props=false] - 默认的开关状态
 * @return {UseSwitchResult}
 * 
 * 
 */
const useSwitch = (props=false) => {
  const [value,setValue] = useState(props);
  const [count,setCount] = useState(props?1:0);
  const refValue = useRef();
  const refCount= useRef();

  useEffect(()=>{
    refValue.current=value;
  },[value]);

  useEffect(()=>{
    refCount.current=count;
  },[count]);

  const actions = useMemo(() => {
    const toggle=()=>{
      setCount(!refValue.current?1:0);
      setValue(!refValue.current);
    };
    const close=(force=false)=>{
      if(force){
        setValue(false);
        setCount(0);
      }else{
        let temp = refCount.current;
        temp = temp-1;
        temp = temp<0?0:temp;
        setCount(temp);
        if(temp>0){
          return;
        }
        setValue(false);
      }
    }
    const open=(force=false)=>{
      if(force){
        setCount(1);
      }else{
        setCount(refCount.current+1);
      }
      setValue(true);
      
    }
    return {toggle,open,close}
  }, []);

  return {state:value,count,...actions};
};

/**
 * useSwitchHookResult
 * @typedef {Object} UseSwitchResult
 * @property {boolean} state - 当前开关状态
 * @property {number} count - 当前开关计数器
 * @property {function} open - 切换至开模式 (force?:boolean)=>void
 * @property {function} close - 切换至关模式 (force?:boolean)=>void
 * @property {function} toggle - 切换至反向模式 ()=>void
 * 
 */


export default useSwitch;
